model = KPrototypes(
    n_clusters=8,
    initialization="frequency",
    numerical_distance="manhattan",
    categorical_distance="matching",
    gamma=None,
    n_iterations=100,
    random_state=None,
    verbose=1,
)

# Fit dataset
clustership = model.fit_predict(numerical_values, categorical_values)
